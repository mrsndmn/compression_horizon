\begin{algorithm}[H]
\caption{Прогрессивная компрессия (progressive cramming)}
\label{alg:progressive}
\begin{algorithmic}[1]
\Require целевая последовательность $x = (x_1, \dots, x_n)$;
         замороженная модель $M$;
         начальная длина $k_0$;
         шаг наращивания $s$;
         бюджет шагов на стадию $T$
\Ensure достигнутая длина компрессии $n_{\mathrm{cram}}$ и эмбеддинг $E^*$
\State $k \gets k_0$
\State инициализировать $E^{(k)}$ случайно
\State $n_{\mathrm{cram}} \gets 0$
\Loop
    \State $x^{(k)} \gets (x_1, \dots, x_k)$
    \State оптимизировать $E^{(k)}$ по задаче~(1)--(2), не более $T$ шагов
    \If{$A_{\mathrm{tf}}\bigl(E^{(k)};\, x^{(k)},\, M\bigr) = 1$}
        \Comment{стадия сошлась}
        \State $E^* \gets E^{(k)}$;\quad $n_{\mathrm{cram}} \gets k$
        \If{$k = n$}
            \State \textbf{break}
            \Comment{достигнут конец последовательности}
        \EndIf
        \State $k' \gets \min(k + s,\; n)$
        \State $E^{(k')} \gets E^{(k)}$
        \Comment{warm-start: перенос решения предыдущей стадии}
        \State $k \gets k'$
    \Else
        \State \textbf{break}
        \Comment{граница успеха/провала достигнута}
    \EndIf
\EndLoop
\State \Return $n_{\mathrm{cram}},\; E^*$
\end{algorithmic}
\end{algorithm}
